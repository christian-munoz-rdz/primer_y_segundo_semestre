ORG 0100H ;CS:IP

MOV AH, 0   ;CONFIGURAR VIDEO
MOV AL, 00H ;40X25 16 COLORES
INT 10H

MOV DX, OFFSET msje ;CARGA MENSAJE EN DX
MOV AH, 9           ;CONFIGURACION PARA DESPLEGAR CADENA
INT 21H
MOV AH, 1           ;CONFIGURACION PARA LEER CARACTER
INT 21H
SUB AL, 30H	    ;DATO INGRESADO EN AL SE CONVIERTE A HEX
MOV BL, AL          ;GUARDA EL PRIMER DATO EN OTRO REGISTRO

MOV AH, 0EH         ;CONFIGURACION PARA IMPRIMIR UN SOLO CARACTER
MOV AL, 10          ;SALTO DE LINEA
INT 10H
MOV AH, 0EH         
MOV AL, 13          ;RETORNO DE CARRO
INT 10H

MOV DX, OFFSET msje2 ;MENSAJE PARA PEDIR EL OTRO NUMERO
MOV AH, 9 
INT 21H
MOV AH, 1           
INT 21H
SUB AL, 30H
MOV CL, AL

MOV AH, 0EH	;SEGUNDO SALTO DE LINEA Y RETORNO        
MOV AL, 10         
INT 10H
MOV AH, 0EH         
MOV AL, 13     
INT 10H

ADD CL,BL 	     ;REALIZA SUMA DE LOS DOS NUMEROS
ADD CL, 30H          ;CONVIERTE EL RESULTADO A CARACTER EN ASCII PARA PODER IMPRIMIRLO

MOV DX, OFFSET msje3 ;IMPRIME MENSAJE CON RESULTADO DE LA SUMA
MOV AH, 9           
INT 21H
MOV AH, 0EH          ;IMPRIME EL CARACTER CON EL RESULTADO DE LA SUMA        
MOV AL, CL           
INT 10H
         

RET

msje db "Dame un numero: $"
msje2 db "Dame otro numero: $"
msje3 db "La suma de los numeros es: $"
